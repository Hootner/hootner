JavaScriptfunction tokenizer(input) {
let current = 0, tokens = [];
while (current < input.length) {
let char = input[current];
if (char === '(') { tokens.push({ type: 'paren', value: '(' }); current++; continue; }
if (char === ')') { tokens.push({ type: 'paren', value: ')' }); current++; continue; }
if (/[a-z]/i.test(char)) {
let value = '';
while (/[a-z]/i.test(char)) { value += char; char = input[++current]; }
tokens.push({ type: 'name', value });
continue;
}
if (/\d/.test(char)) {
let value = '';
while (/\d/.test(char)) { value += char; char = input[++current]; }
tokens.push({ type: 'number', value });
continue;
}
current++;
}
return tokens;
}

function parser(tokens) {
let current = 0;
function walk() {
let token = tokens[current];
if (token.type === 'number') { current++; return { type: 'NumberLiteral', value: token.value }; }
if (token.type === 'paren' && token.value === '(') {
token = tokens[++current];
let node = { type: 'CallExpression', name: token.value, params: [] };
token = tokens[++current];
while (token.type !== 'paren' || (token.type === 'paren' && token.value !== ')')) {
node.params.push(walk());
token = tokens[current];
}
current++;
return node;
}
throw new TypeError(token.type);
}
let ast = { type: 'Program', body: [] };
while (current < tokens.length) ast.body.push(walk());
return ast;
}

function codeGenerator(node) {
switch (node.type) {
case 'Program': return node.body.map(codeGenerator).join('\n');
case 'CallExpression': return node.name + '(' + node.params.map(codeGenerator).join(', ') + ')';

}

function tokenizer(input) {
let current = 0, tokens = [];
while (current < input.length) {
let char = input[current];
if (char === '(') { tokens.push({ type: 'paren', value: '(' }); current++; continue; }
if (char === ')') { tokens.push({ type: 'paren', value: ')' }); current++; continue; }
if (/[a-z]/i.test(char)) {
let value = '';
while (/[a-z]/i.test(char)) { value += char; char = input[++current]; }
tokens.push({ type: 'name', value });
continue;
}
if (/\d/.test(char)) {
let value = '';
while (/\d/.test(char)) { value += char; char = input[++current]; }
tokens.push({ type: 'number', value });
continue;
}
current++;
}
return tokens;
}

function parser(tokens) {
let current = 0;
function walk() {
let token = tokens[current];
if (token.type === 'number') { current++; return { type: 'NumberLiteral', value: token.value }; }
if (token.type === 'paren' && token.value === '(') {
token = tokens[++current];
let node = { type: 'CallExpression', name: token.value, params: [] };
token = tokens[++current];
while (token.type !== 'paren' || token.value !== ')') {
node.params.push(walk());
token = tokens[current];
}
current++;
return node;
}
}
let ast = { type: 'Program', body: [] };
while (current < tokens.length) ast.body.push(walk());
return ast;
}

function codeGenerator(node) {
switch (node.type) {
case 'Program': return node.body.map(codeGenerator).join('\n');
case 'CallExpression': return node.name + '(' + node.params.map(codeGenerator).join(', ') + ')';
case 'NumberLiteral': return node.value;
}
}

function compiler(input) {
const tokens = tokenizer(input);
const ast = parser(tokens);
return codeGenerator(ast);
}

#!/usr/bin/env node
/\*\*

- Smart Dev Journal - Combined journal system
- Auto-captures: commits, errors, fixes, patterns
- Features: daily logs, summaries, pattern tracking, Q training
  \*/

import fs from 'fs/promises';
import path from 'path';
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

const JOURNAL_DIR = 'docs/journal';
const PATTERNS_FILE = 'docs/journal/patterns.json';
const Q_TRAINING_FILE = '.amazonq/training/learned-patterns.md';

class SmartJournal {
constructor() {
this.today = new Date().toISOString().split('T')[0];
this.patterns = { bugs: {}, fixes: {}, correlations: [] };
}

async init() {
await fs.mkdir(JOURNAL_DIR, { recursive: true });
await fs.mkdir('.amazonq/training', { recursive: true });
await this.loadPatterns();
}

async loadPatterns() {
try {
const data = await fs.readFile(PATTERNS_FILE, 'utf-8');
this.patterns = JSON.parse(data);
} catch (e) {
// New patterns file
}
}

async savePatterns() {
await fs.writeFile(PATTERNS_FILE, JSON.stringify(this.patterns, null, 2));
}

// Auto-log from git commits
async autoLogCommits() {
try {
const { stdout } = await execAsync('git log --since="midnight" --pretty=format:"%H|%s|%an|%ad" --date=format:"%H:%M"');
if (!stdout) return [];

      const commits = stdout.split('\n').map(line => {
        const [hash, msg, author, time] = line.split('|');
        return { hash: hash.slice(0, 7), msg, author, time };
      });

      for (const commit of commits) {
        await this.logEntry(commit.msg, 'commit', commit);
      }

      console.log(`✅ Logged ${commits.length} commits`);
      return commits;
    } catch (e) {
      return [];
    }

}

// Smart categorization
categorize(message) {
const lower = message.toLowerCase();

    if (lower.match(/fix|bug|error|issue|crash/)) return '🐛 Bug Fix';
    if (lower.match(/add|create|new|implement/)) return '✨ Feature';
    if (lower.match(/refactor|clean|improve|optimize/)) return '♻️ Refactor';
    if (lower.match(/test|spec|coverage/)) return '🧪 Test';
    if (lower.match(/doc|readme|comment/)) return '📝 Docs';
    if (lower.match(/security|vuln|cve/)) return '🔒 Security';
    if (lower.match(/perf|speed|fast/)) return '⚡ Performance';
    if (lower.match(/deploy|release|ship/)) return '🚀 Deploy';

    return '📌 Task';

}

// Extract patterns for Q training
extractPatterns(message, type) {
const lower = message.toLowerCase();

    // Bug patterns
    if (type === '🐛 Bug Fix') {
      const bugMatch = lower.match(/(syntax|undefined|null|reference|type|async|promise|timeout|memory|leak)/);
      if (bugMatch) {
        const bugType = bugMatch[1];
        this.patterns.bugs[bugType] = (this.patterns.bugs[bugType] || 0) + 1;
      }
    }

    // Fix patterns
    const fixMatch = lower.match(/(added|removed|changed|updated|fixed|replaced) (\w+)/);
    if (fixMatch) {
      const [, action, target] = fixMatch;
      const key = `${action}_${target}`;
      this.patterns.fixes[key] = (this.patterns.fixes[key] || 0) + 1;
    }

    // Correlations (bug -> fix)
    if (type === '🐛 Bug Fix') {
      this.patterns.correlations.push({
        date: this.today,
        bug: message,
        timestamp: new Date().toISOString()
      });
    }

}

// Log entry
async logEntry(message, source = 'manual', metadata = {}) {
const timestamp = new Date().toLocaleTimeString('en-US', { hour12: false, hour: '2-digit', minute: '2-digit' });
const category = this.categorize(message);
const short = message.length > 60 ? message.slice(0, 57) + '...' : message;

    this.extractPatterns(message, category);

    const journalFile = path.join(JOURNAL_DIR, `${this.today}.md`);

    let content = '';
    try {
      content = await fs.readFile(journalFile, 'utf-8');
    } catch (e) {
      content = `# Dev Journal - ${this.today}\n\n`;
    }

    const entry = `- **${timestamp}** ${category} - ${short}\n`;
    content += entry;

    await fs.writeFile(journalFile, content);
    await this.savePatterns();
    await this.updateQTraining();

    console.log(`✅ Logged: ${category} - ${short}`);
    return { timestamp, category, message: short };

}

// Generate Q training data
async updateQTraining() {
const topBugs = Object.entries(this.patterns.bugs)
.sort((a, b) => b[1] - a[1])
.slice(0, 10);

    const topFixes = Object.entries(this.patterns.fixes)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 10);

    const training = `# Learned Patterns for Amazon Q

## Common Bug Types (Auto-Detected)

${topBugs.map(([bug, count]) => `- **${bug}\*\* errors: ${count} occurrences`).join('\n')}

## Common Fix Patterns

${topFixes.map(([fix, count]) => `- **${fix}\*\*: ${count} times`).join('\n')}

## Bug-Fix Correlations

Recent patterns show:
${this.patterns.correlations.slice(-5).map(c => `- ${c.date}: ${c.bug.slice(0, 60)}`).join('\n')}

## Recommendations for Q

When you see:
${topBugs.slice(0, 3).map(([bug]) => `- "${bug}" errors → Check async/await, null checks, type validation`).join('\n')}

---

_Auto-generated: ${new Date().toISOString()}_
_Total bugs tracked: ${Object.values(this.patterns.bugs).reduce((a, b) => a + b, 0)}_
_Total fixes tracked: ${Object.values(this.patterns.fixes).reduce((a, b) => a + b, 0)}_
`;

    await fs.writeFile(Q_TRAINING_FILE, training);

}

const journalFile = path.join(JOURNAL_DIR, `${this.today}.md`);
try {
const content = await fs.readFile(journalFile, 'utf-8');
console.log(content);

console.log(`📝 No entries for ${this.today} yet`);

async viewAll() {
try {
const files = await fs.readdir(JOURNAL_DIR);
const journals = files.filter(f => f.endsWith('.md')).sort().reverse();

      console.log('\n📚 Last 7 Days:\n');

      for (const file of journals.slice(0, 7)) {
        const content = await fs.readFile(path.join(JOURNAL_DIR, file), 'utf-8');
        console.log(content);
        console.log('\n---\n');
      }
    } catch (error) {
      console.error('❌ Error viewing journals:', error.message);
    }

}

// Generate weekly summary
async generateSummary() {
try {
const files = await fs.readdir(JOURNAL_DIR);
const journals = files.filter(f => f.endsWith('.md')).sort().reverse();

      let summary = `# Weekly Summary - ${this.today}\n\n`;
      let totalTasks = 0;

      for (const file of journals.slice(0, 7)) {
        const content = await fs.readFile(path.join(JOURNAL_DIR, file), 'utf-8');
        const lines = content.split('\n').filter(l => l.startsWith('- **'));
        summary += `## ${file.replace('.md', '')}\n`;
        summary += `${lines.length} tasks completed\n\n`;
        totalTasks += lines.length;
      }

      summary += `\n**Total: ${totalTasks} tasks this week**\n`;

      const summaryFile = path.join(JOURNAL_DIR, `summary-${this.today}.md`);
      await fs.writeFile(summaryFile, summary);

      console.log(`✅ Summary saved: ${summaryFile}`);
      console.log(summary);
    } catch (error) {
      console.error('❌ Error generating summary:', error.message);
    }

}

const totalBugs = Object.values(this.patterns.bugs).reduce((a, b) => a + b, 0);
const totalFixes = Object.values(this.patterns.fixes).reduce((a, b) => a + b, 0);
const topBug = Object.entries(this.patterns.bugs).sort((a, b) => b[1] - a[1])[0];

    console.log(`

📊 Journal Statistics

🐛 Total bugs tracked: ${totalBugs}
🔧 Total fixes tracked: ${totalFixes}
🔥 Most common bug: ${topBug ? `${topBug[0]} (${topBug[1]}x)` : 'None yet'}
📈 Correlations tracked: ${this.patterns.correlations.length}

🤖 Q Training: .amazonq/training/learned-patterns.md
📝 Patterns DB: docs/journal/patterns.json
`);
}
}
